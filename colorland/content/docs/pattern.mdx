import Graph from "../../graph";
import { stripe, ring, grid } from "./pattern";

## Stripe Pattern

Example of a stripe pattern.

```js
const layers = [
  [
    {
      x: 120,
      y: 90,
      postRenderCallback(ctx) {
        stripe.drawStripe(ctx, {
          width: 100,
          height: 100,
          lineWidth: 1,
          margin: 9,
          offset: -100,
          segements: [10, 6],
          lineColor: "#ccf",
        });
      }
    },
  ],
];
graph.updateQueue(0, layers);
```

<Graph layers={stripe} options={[]} height={400} styles={{ marginBottom: "32px" }} />
